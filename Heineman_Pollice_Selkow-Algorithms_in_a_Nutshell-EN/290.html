Voronoi Python implementation from heapq import heappop, heappush class Voronoi: def process (self, points): self.pq = [] self.edges = [] self.tree = None self.firstPoint = None # handle tie breakers with first point self.stillOnFirstRow = True self.points = [] # Each point has unique identifier for idx in range(len(points)): pt = Point (points[idx], idx) self.points.append (pt) event = Event (pt, site=pt) heappush (self.pq, event) while self.pq: event = heappop (self.pq) if event.deleted: continue self.sweepPt = event.p if event.site: self.processSite (event) else: self.processCircle (event) # complete edges that remain and stretch to infinity if self.tree and not self.tree.isLeaf: self.finishEdges (self.tree) # Complete Voronoi Edges with partners for e in self.edges: if e.partner: if e.b is None: e.start.y = self.height else: e.start = e.partner.end The implementation handles the special case when there are multiple points that share the same largest y coordinate value; it does so by storing the firstPoint detected within processSite.
